const http = require('http'); const fs = require('fs'); const path = require('path'); const PORT = process.env.PORT || 3000; const DB_FILE = path.join(__dirname, 'leaderboard.json'); // Load or init leaderboard let leaderboard = []; if (fs.existsSync(DB_FILE)) { try { leaderboard = JSON.parse(fs.readFileSync(DB_FILE, 'utf8')); } catch (e) { leaderboard = []; } } function save() { fs.writeFileSync(DB_FILE, JSON.stringify(leaderboard, null, 2)); } function sortBoard() { leaderboard.sort((a, b) => b.score - a.score); } function cors(res) { res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); } function json(res, data, status = 200) { cors(res); res.writeHead(status, { 'Content-Type': 'application/json' }); res.end(JSON.stringify(data)); } const server = http.createServer((req, res) => { cors(res); if (req.method === 'OPTIONS') { res.writeHead(204); res.end(); return; } const url = new URL(req.url, `http://localhost:${PORT}`); // GET /api/leaderboard — top 100 if (req.method === 'GET' && url.pathname === '/api/leaderboard') { sortBoard(); const top = leaderboard.slice(0, 100).map((e, i) => ({ ...e, rank: i + 1 })); return json(res, top); } // GET /api/leaderboard/rank?score=N — what rank would this score get if (req.method === 'GET' && url.pathname === '/api/leaderboard/rank') { const score = parseInt(url.searchParams.get('score') || '0'); sortBoard(); let rank = 1; for (const entry of leaderboard) { if (score >= entry.score) break; rank++; } return json(res, { rank, total: leaderboard.length }); } // POST /api/leaderboard — submit score if (req.method === 'POST' && url.pathname === '/api/leaderboard') { let body = ''; req.on('data', chunk => { body += chunk; }); req.on('end', () => { try { const data = JSON.parse(body); const entry = { name: String(data.name || 'Anonymous').slice(0, 20), score: parseInt(data.score) || 0, publications: parseInt(data.publications) || 0, grants: parseInt(data.grants) || 0, citations: parseInt(data.citations) || 0, hIndex: parseInt(data.hIndex) || 0, mentored: parseInt(data.mentored) || 0, level: String(data.level || '').slice(0, 40), date: new Date().toISOString(), }; leaderboard.push(entry); sortBoard(); save(); const rank = leaderboard.indexOf(entry) + 1; return json(res, { rank, total: leaderboard.length }, 201); } catch (e) { return json(res, { error: 'Invalid JSON' }, 400); } }); return; } json(res, { error: 'Not found' }, 404); }); server.listen(PORT, () => { console.log(`Leaderboard server running on http://localhost:${PORT}`); console.log(` GET /api/leaderboard — top 100`); console.log(` GET /api/leaderboard/rank?score=N`); console.log(` POST /api/leaderboard — submit {name,score,...}`); });
